<script>
  // Function to load content into the main area
  function loadContent(pageName) {
    // Hide all content sections first
    document.querySelectorAll("#content-area > div").forEach(div => {
      div.classList.add("hidden");
    });
    // Show the requested content section
    const targetDiv = document.getElementById(pageName);
    if (targetDiv) {
      targetDiv.classList.remove("hidden");
    }

    // Update active state in sidebar
    document.querySelectorAll("#sidebar a").forEach(link => {
      link.classList.remove("bg-gray-700", "text-white");
      link.classList.add("text-gray-300", "hover:bg-gray-700", "hover:text-white");
    });
    const activeLink = document.querySelector(`#sidebar a[data-page="${pageName}"]`);
    if (activeLink) {
      activeLink.classList.add("bg-gray-700", "text-white");
      activeLink.classList.remove("text-gray-300", "hover:bg-gray-700", "hover:text-white");
    }
  }

  // Initial load of the dashboard overview
  document.addEventListener("DOMContentLoaded", function() {
    loadContent("DashboardOverview");

    // Add event listeners for sidebar navigation
    document.getElementById("sidebar").addEventListener("click", function(event) {
      const target = event.target.closest("a[data-page]");
      if (target) {
        event.preventDefault();
        const pageName = target.getAttribute("data-page");
        loadContent(pageName);
      }
    });

    // Toggle sidebar functionality (optional, if you want to implement it)
    const toggleSidebarButton = document.getElementById("toggle-sidebar");
    const sidebar = document.getElementById("sidebar");
    if (toggleSidebarButton && sidebar) {
      toggleSidebarButton.addEventListener("click", function() {
        sidebar.classList.toggle("hidden"); // Or adjust width with Tailwind classes
      });
    }
  });

  // Placeholder functions for backend calls (to be implemented in Google Apps Script)
  function exportReportAsPdf(reportId) {
    google.script.run.withSuccessHandler(function(url) {
      if (url) {
        window.open(url, "_blank");
      } else {
        alert("Failed to generate PDF report.");
      }
    }).exportReportAsPdf(reportId);
  }

  function exportReportAsExcel(reportId) {
    google.script.run.withSuccessHandler(function(url) {
      if (url) {
        window.open(url, "_blank");
      } else {
        alert("Failed to generate Excel report.");
      }
    }).exportReportAsExcel(reportId);
  }
</script>
